# LBM仿真配置系统设计

**目标**: 通过配置文件驱动不同仿真场景，无需修改核心代码

---

## 1. 配置文件格式（YAML）

### 1.1 示例配置文件

```yaml
# config/laser_melting_ti6al4v.yaml
simulation:
  name: "Ti6Al4V激光熔化 - Phase 5"
  output_dir: "results/ti6al4v_melting"

domain:
  grid: [80, 80, 40]           # nx, ny, nz
  size: [160e-6, 160e-6, 80e-6]  # Lx, Ly, Lz (米)

time:
  dt: 5e-10                     # 时间步长 (秒)
  n_steps: 8000                 # 总步数
  output_interval: 100          # 输出间隔

material:
  type: "Ti6Al4V"               # 材料类型
  # 或自定义:
  # density: 4420
  # cp_solid: 670
  # ...

laser:
  power: 1200.0                 # W
  spot_radius: 50e-6            # 米
  absorption: 0.35
  penetration_depth: 15e-6
  position: [80e-6, 80e-6, 0]   # 中心位置

physics:
  thermal:
    enabled: true
    alpha: 1.0e-5               # 热扩散系数

  phase_change:
    enabled: true
    method: "enthalpy"          # enthalpy / phase_field

  fluid:
    enabled: true
    viscosity: 5e-7             # m²/s

  buoyancy:
    enabled: true
    gravity: [0, 0, 9.81]       # m/s²
    beta_thermal: 9e-6          # K⁻¹
    T_ref: "melting_point"      # 或具体温度值
    force_magnitude: 2e-2       # 格子单位

  darcy_damping:
    enabled: true
    constant: 15.0              # kg/(m³·s)

  marangoni:
    enabled: false              # Phase 6

boundary_conditions:
  thermal:
    type: "adiabatic"           # adiabatic / periodic / fixed
    value: 300.0                # K (for fixed BC)

  fluid:
    type: "no_slip"             # no_slip / free_slip / periodic

initial_conditions:
  temperature: 300.0            # K
  velocity: [0, 0, 0]           # m/s

output:
  format: "vtk"                 # vtk / hdf5
  fields:
    - temperature
    - liquid_fraction
    - velocity
    - phase_state
```

### 2.2 核心类设计

```cpp
// include/config/simulation_config.h
namespace lbm {
namespace config {

struct DomainConfig {
    int nx, ny, nz;
    double Lx, Ly, Lz;
};

struct TimeConfig {
    double dt;
    int n_steps;
    int output_interval;
};

struct MaterialConfig {
    std::string type;  // "Ti6Al4V", "SS316L", "Custom"
    // 如果type="Custom"，使用以下值
    double density;
    double cp_solid, cp_liquid;
    double k_solid, k_liquid;
    double T_solidus, T_liquidus;
    double latent_heat;
    double viscosity;
};

struct LaserConfig {
    bool enabled;
    double power;
    double spot_radius;
    double absorption;
    double penetration_depth;
    std::array<double, 3> position;
};

struct PhysicsConfig {
    bool thermal_enabled;
    bool phase_change_enabled;
    bool fluid_enabled;
    bool buoyancy_enabled;
    bool darcy_enabled;
    bool marangoni_enabled;

    double thermal_alpha;
    double fluid_viscosity;

    std::array<double, 3> gravity;
    double beta_thermal;
    double T_ref;  // -1 表示使用熔点
    double force_magnitude;
    double darcy_constant;
};

struct SimulationConfig {
    std::string name;
    std::string output_dir;

    DomainConfig domain;
    TimeConfig time;
    MaterialConfig material;
    LaserConfig laser;
    PhysicsConfig physics;

    // 边界条件
    std::string thermal_bc;  // "adiabatic", "periodic", "fixed"
    std::string fluid_bc;    // "no_slip", "periodic"
    double thermal_bc_value;

    // 初始条件
    double initial_temperature;

    // 输出
    std::string output_format;
    std::vector<std::string> output_fields;

    // 从YAML加载
    static SimulationConfig loadFromYAML(const std::string& filename);
};

}} // namespace lbm::config
```

## 3. 使用方式

### 3.1 命令行

```bash
# 方式1: 指定配置文件
./run_simulation config/laser_melting_ti6al4v.yaml

# 方式2: 使用预设场景
./run_simulation --preset ti6al4v_melting

# 方式3: 命令行覆盖参数
./run_simulation config/base.yaml --laser-power 1500 --n-steps 10000
```

### 3.2 主程序结构

```cpp
// apps/run_simulation.cu
int main(int argc, char** argv) {
    // 1. 解析命令行参数
    std::string config_file = parseArgs(argc, argv);

    // 2. 加载配置
    auto cfg = lbm::config::SimulationConfig::loadFromYAML(config_file);

    // 3. 打印配置摘要
    cfg.printSummary();

    // 4. 初始化模拟器
    lbm::Simulator sim(cfg);

    // 5. 运行
    sim.initialize();
    sim.run();
    sim.finalize();

    return 0;
}
```

### 3.3 Simulator类（统一入口）

```cpp
// include/core/simulator.h
class Simulator {
public:
    Simulator(const config::SimulationConfig& cfg);

    void initialize();
    void run();
    void finalize();

private:
    config::SimulationConfig cfg_;

    // 物理模块（按需创建）
    std::unique_ptr<physics::ThermalLBM> thermal_;
    std::unique_ptr<physics::FluidLBM> fluid_;
    std::unique_ptr<physics::LaserSource> laser_;

    void setupPhysicsModules();
    void timeStep(int step);
    void output(int step);
};
```
